1. HTML - Add Audio Element

<!-- Add this to your HTML --> <audio id="backgroundMusic" loop preload="auto" autoplay muted> <source src="your-music-file.mp3" type="audio/mpeg"> <source src="your-music-file.ogg" type="audio/ogg"> Your browser does not support the audio element. </audio>

2. JavaScript - Auto-Play Implementation

// Add to your existing JavaScript or create new script const backgroundMusic = document.getElementById('backgroundMusic'); // Set initial volume (0.0 to 1.0) backgroundMusic.volume = 0.3; // 30% volume // Try to play immediately when page loads window.addEventListener('load', () => { backgroundMusic.play().then(() => { console.log('Music started playing automatically'); }).catch((error) => { console.log('Auto-play blocked:', error); // Show user interaction prompt showMusicPrompt(); }); }); // Enable music on first user interaction document.addEventListener('click', enableMusic, { once: true }); document.addEventListener('touchstart', enableMusic, { once: true }); function enableMusic() { backgroundMusic.muted = false; backgroundMusic.play(); } function showMusicPrompt() { // You can show a button or notification here console.log('Click anywhere to enable music'); }
